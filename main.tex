\section*{1. Analytická časť}

\subsection*{(a) Účelová funkcia}
\[
\min \left\{ J(x) =
-\sum_{i=1}^{m} v^i \ln \left( g(x^T u^i) \right)
+ (1 - v^i) \ln \left( 1 - g(x^T u^i) \right)
\;\middle|\; x \in \mathbb{R}^5 \right\}
\]
Dosadíme sigmoid:
\[
g(z) = \frac{1}{1 + e^{-z}}
\]

\[
\ln \left( g(x^T u^i) \right)
= \ln \left( \frac{1}{1 + e^{-x^T u^i}} \right)
= -\ln \left( 1 + e^{-x^T u^i} \right)
\]

\[
\ln \left( 1 - g(x^T u^i) \right)
= \ln \left( \frac{e^{-x^T u^i}}{1 + e^{-x^T u^i}} \right)
= \ln \left( e^{-x^T u^i} \right)
- \ln \left( 1 + e^{-x^T u^i} \right)
\]

Dosadíme do sumy:
\[
v^i \left(-\ln(1 + e^{-x^T u^i})\right)
+ (1 - v^i)\left( -x^T u^i - \ln(1 + e^{-x^T u^i}) \right)
\]

Uprava:
\[
(1 - v^i)x^T u^i + \ln(1 + e^{-x^T u^i})
\]

Výsledná funkcia:
\[
\min \left\{ J(x) =
\sum_{i=1}^{m} \left[(1 - v^i)x^T u^i + \ln(1 + e^{-x^T u^i})\right]
\;\middle|\; x \in \mathbb{R}^5 \right\}
\]


\subsection*{(b) Gradient}
\[ \frac{\partial J}{\partial x_j}=\sum_{i=1}^{m}(x^Tu^i-v^ix^Tu^i+\ln(1+e^{-x^Tu^i}))'=\sum_{i=1}^{m}(u_j^i - v^iu_j^i+\frac{e^{-x^Tu^i}}{1+e^{-x^Tu^i}}(-u_j^i)) \]

\section*{2. Výsledky experimentov – gradientné metódy}

x_{\text{opt}} \approx

\subsection*{(c) Gradientná metóda s konštantným krokom}

\begin{itemize}
    \item \textbf{Výsledok:} $x_c \approx (-1.47360963,\; 3.07175527,\; 10.1989855,\; 17.6905539,\; 0.00484668)$
    \item \textbf{Chyba:} $\|x_c - x_{\text{opt}}\|_2 \approx 0.01472689$
    \item \textbf{Počet iterácií:} $2\,736\,859$
    \item \textbf{Čas:} $66.4\,\mathrm{s}$
\end{itemize}

Metóda je stabilná, ale pomalá. Počet iterácií je veľmi vysoký, čo je dôsledok pevného kroku bez adaptácie.

\subsection*{(d) Backtracking line search}

\begin{itemize}
    \item \textbf{Výsledok:} $x_d \approx (-1.47361102,\; 3.07175538,\; 10.1989558,\; 17.6906325,\; 0.00484666)$
    \item \textbf{Chyba:} $\|x_d - x_{\text{opt}}\|_2 \approx 0.01464526$
    \item \textbf{Počet iterácií:} $631\,781$
    \item \textbf{Čas:} $208\,\mathrm{s}$
    \item \textbf{Získané lambdy:} $\lambda_{\min} \approx 5.58 \times 10^{-6}, \quad \lambda_{\max} \approx 6.97 \times 10^{-6}$
\end{itemize}

Backtracking výrazne zrýchľuje konvergenciu oproti konštantnému kroku a stabilizuje výber dĺžky kroku.

\subsection*{(e) Metóda zlatého rezu}

Optimalizácia dĺžky kroku bola realizovaná pomocou metódy zlatého rezu.

\subsection*{Zhrnutie}

Z porovnania metód vyplýva:

\begin{itemize}
    \item Konštantný krok je najpomalší a vyžaduje najviac iterácií.
    \item Backtracking výrazne zlepšuje konvergenciu a stabilitu výberu kroku.
    \item Metóda zlatého rezu je citlivá na voľbu intervalu:
    \begin{itemize}
        \item širší interval vedie k lepšej presnosti a rýchlejšej konvergencii,
        \item úzky interval môže spôsobiť stagnáciu a výrazne horšie výsledky.
    \end{itemize}
\end{itemize}

Experimenty potvrdzujú, že efektivita gradientných metód silne závisí od vhodnej voľby dĺžky kroku a škálovania problému.
